    @property
    def display_tasks(self) -> List[DisplayTaskAlias]:
        """Get a list of display tasks for the alias."""
        if not self.json["DisplayTasks"]:
            return []
        return [DisplayTaskAlias(dt, self._api) for dt in self.json["DisplayTasks"]]
